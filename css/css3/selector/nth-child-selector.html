ul li{
	width:100px;
	height:100px;
	border:5px solid #ccc;
	margin:0 10px;
	float:left;
	font-size:12px;
}
/* 选择第一个li */
.ul-1 li:first-child{
	background-color:black;
	color:white;
}
/* 选择最后一个li */
.ul-1 li:last-child{
	background-color:silver;
}
/* 选择下标为奇数的li */
.ul-2 li:nth-child(odd){
	font-size:16px;
	font-family:"微软雅黑";
	background-color:maroon;
	color:white;
}
/* 选择下标为偶数的li */
.ul-2 li:nth-child(even){
	font-size:14px;
	font-family:"隶书";
	background-color:purple;
	color:white;
}
/* 选择下标为5的li */
.ul-3 li:nth-child(5){
	background-color:gray;
	color:white;
}
/* 选择下标为2*n的li(n默认为0，逐个递增) */
.ul-4 li:nth-child(2n){
	background-color:maroon;
	color:white;
}
/* 选择下标为2*n的li(n默认为0，逐个递增) */
.ul-5 li:nth-child(n+3){
	background-color:fuchsia;
	color:white;
}
/* 选择下标为-n+3的li(n默认为0，逐个递减) */
.ul-6 li:nth-child(-n+3){
	background-color:green;
	color:orange;
}
/* 选择下标为3n+1的li(n默认为0，逐个递增) */
.ul-7 li:nth-child(3n+1){
	background-color:olive;
	color:white;
}
/* 选择下标为3n-2的li(n默认为0，逐个递增) */
.ul-8 li:nth-child(3n-2){
	background-color:navy;
	color:white;
}
/* 选择倒数第三个li */
.ul-9 li:nth-last-child(3){
	background-color:yellowgreen;
	color:white;
}
/* 从后往前倒序选择3n-2的li(n默认为0，逐个递增) */
.ul-10 li:nth-last-child(3n-2){
	background-color:tomato;
	color:white;
}
