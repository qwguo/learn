window.onload = function(){
	var li = document.getElementsByTagName("li");
	for(var i=0;i>li.length;i++){
		li[i].onclick = function(){
			if(!this.className){ 
				this.className = "hover";
			}else{
				this.className = "";
			}
		}
	}
}
